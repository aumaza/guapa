# tabla g_usuarios
create table if not exists g_usuarios(
id int(11) auto_increment primary key,
name varchar(30) binary not null,
user varchar(65) binary not null,
password varchar(74) binary not null,
email varchar(90) not null,
task int(11) not null,
avatar varchar(500),
role int(1) default 1 not null
)ENGINE=InnoDB DEFAULT CHARSET=utf8;


#tabla g_inquilinos
create table if not exists g_inquilinos(
id int(11) auto_increment primary key,
nombre varchar(100) not null,
dni varchar(10) not null,
num_pasaporte varchar(50),
nacionalidad varchar(100) not null,
email varchar(65) not null,
celular varchar(16) not null,
cbu varchar(24) not null,
alias varchar(50),
image varchar(200)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

#tabla g_propietarios
create table if not exists g_propietarios(
id int(11) auto_increment primary key,
nombre varchar(100) not null,
dni varchar(10) not null,
nacionalidad varchar(50) not null,
direccion varchar(100) not null,
celular varchar(16) not null,
email varchar(65) not null,
cbu varchar(24) not null,
alias varchar(50),
image varchar(200)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

#tabla g_propiedades
create table if not exists g_propiedades(
id int(11) auto_increment primary key,
id_propietario int(11) not null,
localidad varchar(50) not null,
barrio varchar(50) not null,
direccion varchar(50) not null,
piso int(11) not null,
dpto varchar(10) not null,
cant_ambientes int(11) not null
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

#tabla g_empresas
create table if not exists g_empresas(
id int(11) auto_increment primary key,
descripcion varchar(50) not null
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

#tabla g_servicios
create table if not exists g_servicios(
id int(11) auto_increment primary key,
descripcion varchar(50) not null
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

#tabla g_pagos_servicios
create table if not exists g_pago_servicios(
id int(11) auto_increment primary key,
id_propiedad int(11) not null,
id_servicio int(11) not null,
id_empresa int(11) not null,
fecha_vencimiento date not null,
fecha_pago date not null,
importe float(10,2) not null,
comprobante_pago varchar(200)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;
